def tags(tags)
    puts 'validating tags...'

    if tags.nil? || tags.empty?
        return false
    end

    tags.each do |tag|
        puts "validating tag: #{tag}"
        temp = Mer::Tag.new({})
        temp.load_existing(tag.to_i)
        if !temp.exists?
            puts 'Tag does not exist.'.freeze
            return false
        end
    end

    return true
end
